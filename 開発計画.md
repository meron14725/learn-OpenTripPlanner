## システム構成（完成イメージ）

```
[クライアント（curl / フロント）]
        │ POST /api/v1/routes
        │ { origin, destination, desired_arrival_time }
        ▼
[FastAPI バックエンド :8000]
  ├─ Geocoder: 駅名 → 緯度経度（主要駅辞書 / Nominatim）
  └─ OTP2 GraphQL クライアント
        │ planConnection(latestArrival=...) ← 到着時刻指定
        ▼
[OTP2 Java サーバー :8080]
  └─ data/ の GTFS + OSM PBF でグラフ構築済み
        │ 経路・時刻データ
        ▼
[FastAPI バックエンド]
        │ レスポンス整形
        ▼
{ "board_station": "高円寺", "depart_at": "08:35", "line": "中央線快速", ... }
```

### #2-3 Geocoder 実装（駅名 → 緯度経度）

**作業内容**:

- [ ] GTFS の `stops.txt` から全駅の座標辞書を自動生成するスクリプト作成
- [ ] backend/services/geocoder.py 作成（自動生成辞書 + geocode 関数）

**背景**: Phase 1 #1-5 で Nominatim が不採用となったため、駅名辞書の充実が必要。GTFS の `stops.txt` には OTP2 が使用する全駅（約2,060駅）の座標が含まれているので、これを自動インポートする。

**Step 1: 駅名辞書自動生成スクリプト**

各 GTFS zip 内の `stops.txt` を読み、駅名→座標の辞書を JSON として出力する。

```python
# scripts/generate_station_dict.py
import csv
import json
import zipfile
from pathlib import Path

def extract_stops_from_gtfs(gtfs_dir: Path) -> dict[str, tuple[float, float]]:
    """GTFS zip ファイル群から全駅の {name: (lat, lon)} を抽出"""
    stations = {}
    for zip_path in gtfs_dir.glob("*.gtfs.zip"):
        with zipfile.ZipFile(zip_path) as zf:
            if "stops.txt" not in zf.namelist():
                continue
            with zf.open("stops.txt") as f:
                reader = csv.DictReader(line.decode("utf-8-sig") for line in f)
                for row in reader:
                    name = row["stop_name"].replace("駅", "").strip()
                    lat = float(row["stop_lat"])
                    lon = float(row["stop_lon"])
                    if name and name not in stations:
                        stations[name] = (lat, lon)
    return stations

if __name__ == "__main__":
    gtfs_dir = Path(__file__).parent.parent / "data"
    stations = extract_stops_from_gtfs(gtfs_dir)
    out_path = Path(__file__).parent.parent / "backend" / "services" / "station_coords.json"
    out_path.parent.mkdir(parents=True, exist_ok=True)
    with open(out_path, "w", encoding="utf-8") as f:
        json.dump(stations, f, ensure_ascii=False, indent=2)
    print(f"✅ {len(stations)} 駅を {out_path} に出力しました")
```

```bash
python scripts/generate_station_dict.py
# → backend/services/station_coords.json が生成される
```

**Step 2: Geocoder 実装**

自動生成した JSON 辞書を読み込んで使用する。

**`backend/services/geocoder.py`**:

```python
import json
from pathlib import Path

_COORDS_FILE = Path(__file__).parent / "station_coords.json"

def _load_station_coords() -> dict[str, tuple[float, float]]:
    with open(_COORDS_FILE, encoding="utf-8") as f:
        raw = json.load(f)
    return {name: tuple(coords) for name, coords in raw.items()}

STATION_COORDS = _load_station_coords()

def geocode(station_name: str) -> tuple[float, float]:
    """駅名から (latitude, longitude) を返す。駅名末尾の「駅」は除去する。"""
    name = station_name.replace("駅", "").strip()
    if name not in STATION_COORDS:
        raise ValueError(
            f"未対応の駅名: '{name}' — GTFS データに含まれていない駅です"
        )
    return STATION_COORDS[name]
```

```python
  $toLat: Float!,   $toLon: Float!,
  $arrivalTime: OffsetDateTime!
) {
  planConnection(
    origin:      { location: { coordinate: { latitude: $fromLat, longitude: $fromLon } } }
    destination: { location: { coordinate: { latitude: $toLat,   longitude: $toLon   } } }
    dateTime: { latestArrival: $arrivalTime }
    modes: {
      transit: { transit: [{ mode: RAIL }, { mode: SUBWAY }, { mode: BUS }] }
      direct: [WALK]
    }
    first: 3
  ) {
    edges {
      node {
        start
        end
        numberOfTransfers
        legs {
          mode
          route { shortName longName }
          from  { name }
          to    { name }
          start { scheduledTime }
          end   { scheduledTime }
          interlineWithPreviousLeg
        }
      }
    }
  }
}
"""

async def search_route(
    from_lat: float, from_lon: float,
    to_lat:   float, to_lon:   float,
    arrival_time: datetime,
) -> dict:
    # ISO8601 + JST オフセット付き文字列に変換
    jst = timezone(timedelta(hours=9))
    arrival_str = arrival_time.astimezone(jst).strftime("%Y-%m-%dT%H:%M:%S+09:00")

    async with httpx.AsyncClient(timeout=30) as client:
        resp = await client.post(
            OTP2_GRAPHQL_URL,
            json={
                "query": PLAN_QUERY,
                "variables": {
                    "fromLat": from_lat, "fromLon": from_lon,
                    "toLat":   to_lat,   "toLon":   to_lon,
                    "arrivalTime": arrival_str,
                },
            },
        )
        resp.raise_for_status()
        return resp.json()
```

### #2-5 API エンドポイント実装

**作業内容**:

- [ ] backend/api/routes.py 作成（/api/v1/routes エンドポイント）

**`backend/api/routes.py`**:

```python
from fastapi import APIRouter, HTTPException
from datetime import datetime, timezone, timedelta
from models.request import RouteRequest
from models.response import RouteResponse, TransitLeg
from services.geocoder import geocode
from services.otp_client import search_route

router = APIRouter()
JST = timezone(timedelta(hours=9))


def fmt_time(iso_str: str) -> str:
    """ISO8601 文字列を "HH:MM" (JST) に変換する"""
    return datetime.fromisoformat(iso_str).astimezone(JST).strftime("%H:%M")


@router.post("/api/v1/routes", response_model=RouteResponse)
async def get_routes(req: RouteRequest):
    # 1. 駅名 → 座標
    try:
        from_lat, from_lon = geocode(req.origin)
        to_lat,   to_lon   = geocode(req.destination)
    except ValueError as e:
        raise HTTPException(status_code=400, detail=str(e))

    # 2. OTP2 に経路検索
    data = await search_route(from_lat, from_lon, to_lat, to_lon, req.desired_arrival_time)

    edges = data.get("data", {}).get("planConnection", {}).get("edges", [])
    if not edges:
        raise HTTPException(status_code=404, detail="経路が見つかりませんでした（時刻・区間を確認してください）")

    itinerary = edges[0]["node"]  # 最初の候補を採用

    # 3. legs を整形
    legs = [
        TransitLeg(
            mode=leg["mode"],
            line_name=leg["route"]["shortName"] if leg.get("route") else None,
            from_station=leg["from"]["name"],
            to_station=leg["to"]["name"],
            departure_time=fmt_time(leg["start"]["scheduledTime"]),
            arrival_time=fmt_time(leg["end"]["scheduledTime"]),
        )
        for leg in itinerary["legs"]
    ]

    # 4. 最初の電車区間 = 最寄り駅からの乗車
    first_train = next((l for l in legs if l.mode not in ("WALK",)), legs[0])

    return RouteResponse(
        board_station=first_train.from_station,
        depart_at=first_train.departure_time,
        arrive_at=fmt_time(itinerary["end"]),
        num_transfers=itinerary["numberOfTransfers"],
        legs=legs,
        message=f"{first_train.departure_time} に {first_train.from_station} を出発してください",
    )
```
